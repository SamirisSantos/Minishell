Sinais
bash# prompt vazio:
Ctrl+C         # nova linha com novo prompt
Ctrl+\         # nao faz nada
Ctrl+D         # sai do minishell — relança!

# com texto escrito:
echo hel → Ctrl+C    # limpa buffer
echo hel → Ctrl+D    # nao faz nada
echo hel → Ctrl+\    # nao faz nada

# com comando bloqueante:
cat → Ctrl+C         # termina cat
cat → Ctrl+\         # termina cat
cat → Ctrl+D         # termina cat (EOF)
grep "x" → Ctrl+C
